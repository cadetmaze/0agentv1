```typescript
async runCompaction(): Promise<void> {
  // Safety: never run compaction while active sessions are open
  const active = this.sessionManager.getActiveCount()
  if (active > 0) {
    console.log(`[Compaction] Deferring: ${active} active sessions`)
    return

  // Level 4: LRU eviction — handled automatically by WorkingMemory on each access
```

#### Level 1: Edge Pruning

```typescript
// EdgePruner.ts
async prune(config: PruneConfig): Promise<PruneResult> {
  const now = Date.now()
  const idle_threshold_ms = config.idle_days * 24 * 60 * 60 * 1000
  const result: PruneResult = { pruned_count: 0, skipped_locked: 0,
                                 skipped_hypothesis: 0, skipped_active: 0,
                                 pruned_edge_ids: [] }

  const candidates = this.adapter.getAllEdges(this.graph_id)
  for (const edge of candidates) {
    // Condition: |weight - 0.5| < threshold AND idle for N days AND not locked
    const distFromNeutral = Math.abs(edge.weight - 0.5)
    if (distFromNeutral >= config.weight_delta_threshold) continue

    const lastTraversed = edge.last_traversed ?? edge.created_at
    const idleMs = now - lastTraversed
    if (idleMs < idle_threshold_ms) {
      result.skipped_active++
      continue
    }

    if (edge.locked) {
      result.skipped_locked++
      continue
    }

    // Exception: never prune edges connected to HYPOTHESIS nodes
    const fromNode = this.graph.getNode(edge.from_node)
    const toNode   = this.graph.getNode(edge.to_node)
    if (fromNode?.type === NodeType.HYPOTHESIS || toNode?.type === NodeType.HYPOTHESIS) {
      result.skipped_hypothesis++
      continue
    }

    // Exception: never prune edges connected to skill-type nodes
    // Skill nodes represent workflow structure, not learned data — they must persist
    if (fromNode?.metadata?.is_skill === true || toNode?.metadata?.is_skill === true) {
      result.skipped_skill++
      continue
    }

    if (!config.dry_run) {
      this.adapter.deleteEdge(edge.id)
      // Also delete weight_events for this edge (they're no longer useful)
      this.adapter.deleteWeightEvents(edge.id)
    }
    result.pruned_count++
    result.pruned_edge_ids.push(edge.id)
  }

  return result
}
```

#### Level 3: Node Deduplication

```typescript
// NodeDeduplicator.ts
async run(config: DeduplicationConfig): Promise<DeduplicationResult> {
  // Get all nodes WITH embeddings (only these can be deduped by cosine sim)
  const nodes_with_embeddings = this.adapter.getNodesWithEmbeddings(this.graph_id)

  const candidates: MergeCandidate[] = []
  const seen = new Set<string>()

  // Pairwise cosine similarity — O(n²) but only run weekly, and only on embedded nodes
  // For large graphs (>1000), use HNSW approximate search instead
  for (let i = 0; i < nodes_with_embeddings.length; i++) {
    for (let j = i + 1; j < nodes_with_embeddings.length; j++) {
      const a = nodes_with_embeddings[i]
      const b = nodes_with_embeddings[j]

      // Exception: never merge nodes of different types
      if (a.type !== b.type) continue
      if (seen.has(`${a.id}:${b.id}`)) continue

      const sim = cosineSimilarity(a.embedding!, b.embedding!)
      if (sim > config.cosine_threshold) {
        const surviving = a.visit_count >= b.visit_count ? a : b
        const removed   = surviving.id === a.id ? b : a
        candidates.push({ node_a_id: a.id, node_b_id: b.id, similarity: sim,
                          surviving_node_id: surviving.id, removed_node_id: removed.id })
        seen.add(`${a.id}:${b.id}`)
      }
    }
  }

  let merged_count = 0
  if (!config.dry_run) {
    for (const candidate of candidates) {
      await this.mergeNodes(candidate)
      merged_count++
    }
  }

  return { merged_count, candidates_found: candidates.length, candidates }
}

private async mergeNodes(candidate: MergeCandidate): Promise<void> {
  const surviving = this.graph.getNode(candidate.surviving_node_id)!
  const removed   = this.graph.getNode(candidate.removed_node_id)!

  // Merge strategy (implement exactly):
  // new_label = higher_visit_count node's label (already surviving)
  // new_visit_count = sum
  const new_visit_count = surviving.visit_count + removed.visit_count

  // new_metadata = deep merge (conflicts → surviving wins)
  const new_metadata = { ...removed.metadata, ...surviving.metadata }

  // Update surviving node
  this.adapter.updateNode(candidate.surviving_node_id, {
    visit_count: new_visit_count,
    metadata: new_metadata,
  })

  // Re-point all edges: from_node = removed → surviving
  const outgoing = this.adapter.getEdgesByNode(removed.id, 'from')
  for (const edge of outgoing) {
    // Check if equivalent edge already exists on surviving (same type, same to_node)
    const existing = this.adapter.findEdge(candidate.surviving_node_id, edge.to_node, edge.type)
    if (existing) {
      // Merge weights: take higher weight edge, add traversal counts
      const merged_weight = Math.max(existing.weight, edge.weight)
      const merged_traversals = existing.traversal_count + edge.traversal_count
      this.adapter.updateEdge(existing.id, { weight: merged_weight, traversal_count: merged_traversals })
      this.adapter.deleteEdge(edge.id)
    } else {
      this.adapter.updateEdgeFromNode(edge.id, candidate.surviving_node_id)
    }
  }

  // Re-point: to_node = removed → surviving
  const incoming = this.adapter.getEdgesByNode(removed.id, 'to')
  for (const edge of incoming) {
    const existing = this.adapter.findEdge(edge.from_node, candidate.surviving_node_id, edge.type)
    if (existing) {
      const merged_weight = Math.max(existing.weight, edge.weight)
      this.adapter.updateEdge(existing.id, { weight: merged_weight })
      this.adapter.deleteEdge(edge.id)
    } else {
      this.adapter.updateEdgeToNode(edge.id, candidate.surviving_node_id)
    }
  }

  // Reassign weight_events to surviving node (via edges, already re-pointed above)

  // Delete removed node
  this.adapter.deleteNode(removed.id)
}
```

### Graph Rollback (3 modes, implement exactly)

```typescript
// GraphRollback.ts

async rollback(mode: RollbackMode): Promise<RollbackResult> {
  switch (mode.type) {
    case 'checkpoint': return this.rollbackToCheckpoint(mode.checkpoint_id)
    case 'trace':      return this.rollbackTrace(mode.trace_id)
    case 'verifier':   return this.rollbackVerifier(mode.verifier_name)
  }
}

// Mode 1: By checkpoint
private async rollbackToCheckpoint(checkpoint_id: string): Promise<RollbackResult> {
  const checkpoint = await this.checkpointStore.get(checkpoint_id)
  if (!checkpoint) throw new Error(`Checkpoint ${checkpoint_id} not found`)

  const snapshot: CheckpointSnapshot = JSON.parse(
    await this.objectStore.read(checkpoint.storage_path)
  )

  // Step 1: Replay weight_events BACKWARDS from now to checkpoint timestamp
  // Get all weight_events created AFTER checkpoint timestamp, ordered DESC
  const events_to_reverse = this.adapter.getWeightEventsAfter(checkpoint.created_at)
  events_to_reverse.reverse()  // newest first

  let edges_reverted = 0
  for (const event of events_to_reverse) {
    // Apply inverse: new_weight → old_weight
    const current = this.adapter.getEdge(event.edge_id)
    if (!current) continue
    this.adapter.db.prepare(`UPDATE edges SET weight = ? WHERE id = ?`)
      .run(event.old_weight, event.edge_id)
    edges_reverted++
  }

  // Step 2: Re-create deleted nodes/edges (from checkpoint snapshot)
  // Nodes in snapshot but not in current graph → re-insert
  const current_node_ids = new Set(this.adapter.getAllNodeIds(this.graph.id))
  let nodes_restored = 0
  for (const node of snapshot.nodes) {
    if (!current_node_ids.has(node.id)) {
      this.adapter.insertNode(node)
      nodes_restored++
    }
  }
  // Edges similarly
  const current_edge_ids = new Set(this.adapter.getAllEdgeIds(this.graph.id))
  for (const edge of snapshot.edges) {
    if (!current_edge_ids.has(edge.id)) {
      this.adapter.insertEdge(edge)
    }
  }

  // Step 3: Delete weight_events created after checkpoint (they're now invalid)
  this.adapter.deleteWeightEventsAfter(checkpoint.created_at)

  return { mode: { type: 'checkpoint', checkpoint_id },
           edges_reverted, nodes_restored, traces_undone: 0,
           weight_events_reversed: events_to_reverse.length }
}

// Mode 2: By trace
private async rollbackTrace(trace_id: string): Promise<RollbackResult> {
  // Get all weight_events for this trace, ordered by created_at DESC (newest first)
  const events = this.adapter.getWeightEventsByTrace(trace_id)
  events.sort((a, b) => b.created_at - a.created_at)  // newest first = reverse order

  let edges_reverted = 0
  for (const event of events) {
    // Apply inverse delta: restore old_weight
    const current = this.adapter.getEdge(event.edge_id)
    if (!current) continue
    this.adapter.db.prepare(`UPDATE edges SET weight = ? WHERE id = ?`)
      .run(event.old_weight, event.edge_id)
    edges_reverted++
  }

  // Delete the weight_events (they no longer represent valid history)
  this.adapter.deleteWeightEventsByTrace(trace_id)

  // Mark trace as rolled back in metadata
  this.adapter.db.prepare(`UPDATE traces SET metadata = json_set(metadata, '$.rolled_back', 1) WHERE id = ?`)
    .run(trace_id)

  return { mode: { type: 'trace', trace_id },
           edges_reverted, nodes_restored: 0, traces_undone: 1,
           weight_events_reversed: events.length }
}

// Mode 3: By verifier
private async rollbackVerifier(verifier_name: string): Promise<RollbackResult> {
  // Find all traces resolved by this verifier
  const traces = this.adapter.db.prepare(
    `SELECT id FROM traces WHERE json_extract(metadata, '$.deferred_verifier') = ? AND outcome_type != 'expired'`
  ).all(verifier_name) as { id: string }[]

  let total_edges_reverted = 0
  let total_events = 0
  for (const trace of traces) {
    const r = await this.rollbackTrace(trace.id)
    total_edges_reverted += r.edges_reverted
    total_events += r.weight_events_reversed
  }

  return { mode: { type: 'verifier', verifier_name },
           edges_reverted: total_edges_reverted, nodes_restored: 0,
           traces_undone: traces.length, weight_events_reversed: total_events }
}
```

### OCC Session Snapshot Isolation

```typescript
// SessionSnapshot.ts

class SessionSnapshotManager {
  private snapshots = new Map<string, SessionSnapshot>()

  // Call at session start — captures current edge weights
  createSnapshot(session_id: string, relevant_edge_ids: string[]): SessionSnapshot {
    const snapshot: SessionSnapshot = {
      session_id,
      created_at: Date.now(),
      edges: new Map(),
      pending_updates: new Map(),
    }

    for (const edge_id of relevant_edge_ids) {
      const edge = this.adapter.getEdge(edge_id)
      if (edge) {
        snapshot.edges.set(edge_id, {
          edge_id,
          weight_at_snapshot: edge.weight,
          version: this.getEdgeVersion(edge_id),  // count of weight_events
        })
      }
    }

    this.snapshots.set(session_id, snapshot)
    return snapshot
  }

  // Call at session end — merge session's learnings into live graph
  async mergeSnapshot(session_id: string, session_succeeded: boolean): Promise<MergeResult> {
    const snapshot = this.snapshots.get(session_id)
    if (!snapshot) throw new Error(`No snapshot for session ${session_id}`)

    const result: MergeResult = { applied: 0, conflicted: 0, skipped: 0 }

    for (const [edge_id, pending_weight] of snapshot.pending_updates) {
      const current = this.adapter.getEdge(edge_id)
      if (!current) { result.skipped++; continue }

      const snapshot_entry = snapshot.edges.get(edge_id)
      const current_version = this.getEdgeVersion(edge_id)

      if (snapshot_entry && current_version > snapshot_entry.version) {
        // Another session modified this edge during our session
        result.conflicted++
        if (session_succeeded) {
          // Session learnings take priority: force write
          this.adapter.db.prepare(`UPDATE edges SET weight = ? WHERE id = ?`)
            .run(pending_weight, edge_id)
          result.applied++
        } else {
          // Session failed: discard our update, keep live graph
          result.skipped++
        }
      } else {
        // No conflict: standard OCC update
        const success = this.adapter.updateEdgeWeight(edge_id, pending_weight, current.weight)
        if (success) result.applied++
        else { result.conflicted++; result.skipped++ }
      }
    }

    this.snapshots.delete(session_id)
    return result
  }

  private getEdgeVersion(edge_id: string): number {
    const row = this.adapter.db.prepare(
      `SELECT COUNT(*) as cnt FROM weight_events WHERE edge_id = ?`
    ).get(edge_id) as { cnt: number }
    return row.cnt
  }
}
```

```rust
// graph_traversal.rs

use napi_derive::napi;
use std::collections::{BinaryHeap, HashMap, HashSet, VecDeque};

#[derive(Debug, Clone)]
pub struct EdgeData {
    pub id: String,
    pub from_node: String,
    pub to_node: String,
    pub weight: f64,
}

#[napi(object)]
pub struct PathResult {
    pub node_ids: Vec<String>,
    pub edge_ids: Vec<String>,
    pub weight_product: f64,
}

/// BFS from start_node, return top-k paths by weight product.
/// `max_depth` limits path length.
#[napi]
pub fn bfs_top_k(
    start_node: String,
    edges: Vec<serde_json::Value>,  // passed as JSON from TS
    max_depth: u32,
    top_k: u32,
) -> Vec<PathResult> {
    let edge_map = build_adjacency_map(&edges);

    struct QueueItem {
        node: String,
        path_nodes: Vec<String>,
        path_edges: Vec<String>,
        weight_product: f64,
        depth: u32,
    }

    let mut queue = VecDeque::new();
    queue.push_back(QueueItem {
        node: start_node.clone(),
        path_nodes: vec![start_node],
        path_edges: vec![],
        weight_product: 1.0,
        depth: 0,
    });

    let mut results: Vec<PathResult> = vec![];
    let mut visited: HashSet<String> = HashSet::new();

    while let Some(item) = queue.pop_front() {
        if item.depth >= max_depth { continue; }
        visited.insert(item.node.clone());

        if let Some(neighbors) = edge_map.get(&item.node) {
            for (neighbor_id, edge_id, edge_weight) in neighbors {
                if visited.contains(neighbor_id) { continue; }
                let new_product = item.weight_product * edge_weight;
                let mut new_nodes = item.path_nodes.clone();
                let mut new_edges = item.path_edges.clone();
                new_nodes.push(neighbor_id.clone());
                new_edges.push(edge_id.clone());

                results.push(PathResult {
                    node_ids: new_nodes.clone(),
                    edge_ids: new_edges.clone(),
                    weight_product: new_product,
                });

                queue.push_back(QueueItem {
                    node: neighbor_id.clone(),
                    path_nodes: new_nodes,
                    path_edges: new_edges,
                    weight_product: new_product,
                    depth: item.depth + 1,
                });
            }
        }
    }

    // Sort by weight_product descending, return top-k
    results.sort_by(|a, b| b.weight_product.partial_cmp(&a.weight_product).unwrap());
    results.truncate(top_k as usize);
    results
}

/// Batch decay: returns Vec of (edge_id, new_weight) for all edges that need updating.
#[napi]
pub fn batch_decay(
    edges_json: String,
    now_ms: f64,
    grace_ms: f64,
    decay_rate_default: f64,
    max_delta: f64,
) -> String {
    let edges: Vec<serde_json::Value> = serde_json::from_str(&edges_json).unwrap_or_default();
    let mut updates: Vec<serde_json::Value> = vec![];

    for edge in &edges {
        let locked = edge["locked"].as_bool().unwrap_or(false);
        if locked { continue; }

        let weight = edge["weight"].as_f64().unwrap_or(0.5);
        let last_traversed = edge["last_traversed"].as_f64()
            .unwrap_or_else(|| edge["created_at"].as_f64().unwrap_or(now_ms));
        let decay_rate = edge["decay_rate"].as_f64().unwrap_or(decay_rate_default);

        let age_ms = now_ms - last_traversed;
        if age_ms < grace_ms { continue; }

        let hours_elapsed = age_ms / 3_600_000.0;
        let distance = (weight - 0.5).abs();
        let raw_delta = decay_rate * distance * hours_elapsed;
        let capped_delta = raw_delta.min(max_delta);

        let direction: f64 = if weight > 0.5 { -1.0 } else { 1.0 };
        let new_weight = weight + direction * capped_delta;
        // Clamp — never cross 0.5
        let clamped = if direction > 0.0 {
            new_weight.min(0.5)
        } else {
            new_weight.max(0.5)
        };

        if (clamped - weight).abs() > 1e-6 {
            updates.push(serde_json::json!({
                "edge_id": edge["id"],
                "new_weight": clamped
            }));
        }
    }

    serde_json::to_string(&updates).unwrap()
}
```

```typescript
import { KnowledgeGraph } from '@0agent/core'
import graphData from './graph.json'

export async function installSeed(graph: KnowledgeGraph, graph_id: string): Promise<void> {
  // Check: don't install if seed nodes already present (idempotent)
  const existing = graph.queryNodes(graph_id, undefined, 1)
  if (existing.length > 0) {
    console.log(`[Seed] ${graphData.seed_name} already installed, skipping`)
    return
  }

  // Insert nodes first (edges reference them)
  for (const node of graphData.nodes) {
    await graph.addNode({ ...node, graph_id, created_at: Date.now(), last_seen: Date.now() })
  }

  // Insert edges
  for (const edge of graphData.edges) {
    await graph.addEdge({ ...edge, graph_id, created_at: Date.now(), traversal_count: 0 })
  }

  console.log(`[Seed] Installed ${graphData.seed_name}: ${graphData.nodes.length} nodes, ${graphData.edges.length} edges`)
}
```

## Risks and Gotchas

1. **napi-rs binary compatibility** — napi-rs native modules are compiled against a specific Node.js ABI. If the user runs a different Node version than what the `.node` file was compiled for, it crashes. The fallback loader (`index.js`) MUST catch this and switch to TypeScript gracefully. Always build for multiple targets with `napi-rs cross` and provide prebuilds.

2. **10K trace test memory** — Running 10K traces in a single test process will accumulate a lot of in-memory state. Use `--forceExit` in Vitest and reset `WorkingMemory` between batches of 1000. Use a `PRAGMA auto_vacuum = INCREMENTAL` on SQLite to prevent DB file growth from deleted rows.

3. **Node dedup cosine computation** — Pairwise cosine similarity is O(n²). For 10K nodes with embeddings: ~50M comparisons. Instead, use HNSW approximate search: for each node, find its k nearest neighbors, filter by cosine threshold. This reduces to O(n · k · log(n)). Implement both: exact (for tests with small graphs) and HNSW-accelerated (for production).

4. **Rollback atomicity** — Rollback reverses weight_events one at a time. If the process crashes halfway, the graph is in a corrupt intermediate state. Wrap the entire rollback in a SQLite transaction: `BEGIN IMMEDIATE ... COMMIT`. This is the single most important production safety requirement in Phase 5.

5. **Checkpoint storage size** — A checkpoint snapshot includes ALL nodes and edges as JSON. For a 10K node graph, this can be 50-100MB per checkpoint. Compress with gzip before writing to ObjectStore. Add size budget: refuse to create checkpoint if it would exceed 500MB. Implement checkpoint rotation (keep last N checkpoints, delete older).

6. **Seed graph node ID collisions** — Seed graph nodes have pre-defined string IDs (like `"node_debug_root"`). If two seeds use the same IDs, the second install will silently overwrite the first. Namespace all seed node IDs by seed name: `"seed_se_node_debug_root"`, `"seed_re_node_hypothesis"`, etc. Enforce this in the `install.ts` script.

7. **Python SDK blocking run()** — `agent.run()` polls via HTTP with 1-second sleep. For long tasks (> 5 minutes), this is inefficient but correct. Provide an async version `await agent.run_async(task)` using `asyncio` + `httpx.AsyncClient` for production use. The sync version is fine for scripts and tests.

8. **Rust napi-rs tokio conflict** — If the napi-rs module uses `tokio_rt` feature, and Node.js/Bun also has its own async runtime, there can be thread pool conflicts. For batch_decay (synchronous CPU work), disable `tokio_rt` and use synchronous napi functions. Only use async napi for I/O operations (which batch_decay doesn't need).

9. **OCC conflict < 1% target** — At 10K traces with concurrent sessions, 1% = 100 conflicts. The main driver of conflicts is concurrent weight updates to popular strategy edges (many sessions choosing the same path). Mitigate by: (a) staggered session scheduling (not all at once), (b) exponential backoff in `EdgeWeightUpdater`, (c) LWW fallback on 3rd retry doesn't count as a "conflict" if it succeeds. Log conflict rate to a metric.

10. **A2A auth with bearer token** — The MCP server's bearer token is stored in `~/.0agent/config.yaml`. The connecting agent (Claude Code, other 0agent) must be given this token out-of-band. In the init wizard, after enabling MCP server mode, print: `"Your MCP connection string: 0agent+mcp://localhost:4200?token=<token>"`. This string can be pasted into Claude Desktop or another 0agent's config.

11. **Overlay graph merge complexity** — The overlay model (personal overlay + base graph) requires that `KnowledgeGraph.getEdge()` checks overlay first, then base. But `queryMerged()` must blend results from both, with overlay weights taking priority for matching edges. Implement as: (1) run query against base, (2) for each returned edge, check if overlay has an override, (3) return overlay weight if present. The base graph is read-only from the user's perspective — only overlay writes.

12. **Skill node metadata integrity** — The `is_skill` flag in `node.metadata` is the sole protection mechanism for skill nodes during compaction. If a migration or manual edit strips this flag, skill workflow edges will be pruned on the next compaction cycle. Add a startup integrity check: verify all nodes referenced by `SkillRegistry` have `metadata.is_skill === true`, and repair any missing flags with a warning log.

13. **Custom skill package version conflicts** — If two `@0agent/skill-*` packages define a skill with the same `name`, the second one silently overwrites the first in `SkillRegistry`. Add a conflict detection log: `[SkillRegistry] WARNING: skill 'lint' from @0agent/skill-lint-v2 overwrites existing skill from @0agent/skill-lint`. Consider using package version to break ties (higher version wins).

14. **`deleteWeightEventsAfter` in rollback** — After reversing weights for a checkpoint rollback, you must delete the weight_events that were created after the checkpoint. But those events are the authoritative history. Add a `rollback_events` table (just the event IDs that were reversed) so the rollback itself is auditable and reversible. This is optional for Phase 5 but avoids "we can't undo the rollback" issues in production.

- **Phase 1 (Core)**: `SQLiteAdapter` needs ~8 new methods added. Do NOT modify existing prepared statements — only add new ones. This prevents regressions in Phase 1 tests.
- **Phase 2 (Daemon)**: `BackgroundWorkers.ts` gains compaction scheduling. `CompactionOrchestrator` is injected at daemon startup via `DaemonStartupOptions`. The `compactor_interval_ms` field in `WorkerConfig` (Phase 2 stub) is now used.
- **Phase 3 (Subagents)**: `SessionSnapshot` wraps the subagent spawn: create snapshot before spawning, merge after result received. This ensures subagent outcomes don't race with concurrent sessions.
- **Phase 4 (Learning)**: `TraceReplay.ts` from Phase 4 uses `getWeightEventsAfter()` from Phase 5's schema additions. `CreditAttribution` results feed into the rollback's `weight_events` (each attribution creates a weight_event with `trace_id`). Rollback-by-trace reverses all attribution events for that trace.
- **Seeds and A2A**: Seed graphs ship as npm packages and are installed during `0agent init` (step 5). After Phase 5 is complete, the onboarding wizard's seed step (Phase 2) can actually work. The MCP server mode (Phase 5) makes the Phase 2 MCPHub server stub fully functional.
- **Phase 2 (Skill Library)**: The sprint workflow seed graph (`sprint-workflow.json`) provides initial weights for the skill workflow that Phase 2's `SkillRegistry` models. Custom skill packages (`@0agent/skill-*`) extend the `SkillRegistry.loadAll()` method introduced in Phase 2. The Python SDK's skill methods call the same `/api/skills/*` REST routes that Phase 2's daemon exposes.
- **Phase 3 (Skill Invoker)**: Skill nodes with `metadata.is_skill = true` in the compaction-protected graph are the same nodes that Phase 3's `SkillInvoker` references when building `SpawnRequest.skill`. The `EdgePruner` skill protection ensures workflow structure is never lost.
- **Phase 4 (Learning Pipeline)**: The `/retro` skill's `learning_signal` outcome type (Phase 4) updates the same sprint workflow edge weights that are seeded here. The `WorkflowSuggestionEngine` (Phase 4) reads these edges to recommend next skills.
